          f_{\mathrm{w}}(t)=Acc_{\mathrm{w}}(t)-Acc_{\mathrm{w}}(0)
        \]
        \vspace{-1.05em}
        \[
          \Delta_{\mathrm{tool}}(t)=G(t)-G(0)
        \]
        \vspace{-1.05em}

          =
          \underbrace{f_{\mathrm{wo}}(t)}_{\text{Intrinsic drift}}
